    document.getElementById('ship-fill').style.width = pct + '%';

async function doSM2Payment() {

  const ln   = document.getElementById('sm2-ln')?.value?.trim();
  const ml   = document.getElementById('sm2-mail')?.value?.trim();

  const btn = document.getElementById('sm2-paybtn');
  const lbl = document.getElementById('sm2-paylbl');
  btn.disabled = true;
  lbl.innerHTML = '
 WIRD VERARBEITET...';
  if(errEl) { errEl.style.display='none'; errEl.innerText=''; }

  if(!sm2Elements||!sm2PayEl) {
    btn.disabled=false; lbl.innerText='JETZT SICHER BEZAHLEN';
    toast('Stripe nicht bereit. Seite neu laden.','err'); return;
  }

  try {
    // 1. Stripe Elements validieren
    const { error: subErr } = await sm2Elements.submit();
    if(subErr) throw subErr;

    const sub   = cart.reduce((a,i) => a + i.price*i.qty, 0);
    const disc  = calcDiscount(sub);
    const discSub = sub - disc;
    const ship  = getShipping(land, discSub);
    const total = discSub + ship;
    const orderNr = 'LXRD-' + Date.now().toString().slice(-6);
    const cc    = (land.length===2 ? land : 'DE').toUpperCase();

    // 2. Payment Intent via Stripe (direkt – für GitHub Pages)
    //    WICHTIG: Dieser Aufruf geht an DEINEN Stripe Secret Key Server.
    //    Da GitHub Pages keinen Server hat, leiten wir zu Stripe Checkout weiter.
    //    ─────────────────────────────────────────────────────────────────────────
    //    OPTION A: Stripe Payment Links (empfohlen für GitHub Pages)
    //    Erstelle in Stripe Dashboard einen Payment Link und setze ihn hier ein:
    const STRIPE_PAYMENT_LINK = 'https://buy.stripe.com/DEIN_LINK'; // ← hier eintragen!
    //    ─────────────────────────────────────────────────────────────────────────

    // Bestellnummer speichern vor Weiterleitung

    sessionStorage.setItem('lexord_order_name',    fn+' '+ln);
    sessionStorage.setItem('lexord_order_nr',      orderNr);
    sessionStorage.setItem('lexord_order_land',    land);

    sessionStorage.setItem('lexord_order_disc',    disc.toFixed(2));
    sessionStorage.setItem('lexord_order_ship',    ship.toFixed(2));

    // E-Mail vorab senden (vor Stripe-Redirect)
    sendOrderEmail(ml, orderNr, 'Stripe ('+land+')', {
      payer: { name:{ given_name:fn }, email_address:ml }
    }, fn, ln, str+', '+plz+' '+ort+', '+land);

    // Zum Stripe Payment Link weiterleiten mit Metadaten
    const url = new URL(STRIPE_PAYMENT_LINK);
    url.searchParams.set('prefilled_email', ml);
    url.searchParams.set('client_reference_id', orderNr);

    window.location.href = url.toString();

  } catch(err) {
    console.error('Stripe:', err);
    if(errEl) { errEl.innerText = err.message||'Zahlung fehlgeschlagen. Bitte erneut versuchen.'; errEl.style.display='block'; }

  c.innerHTML = '';

    paypal.Buttons({

        if (!document.getElementById('cart-agb').checked) { toast('Bitte AGB akzeptieren','err'); return a.reject(); }
      },
      createOrder: (d, a) => a.order.create({
        purchase_units: [{
          amount: {
            currency_code:'EUR',
            value: (sub+ship).toFixed(2),
            breakdown: {
              item_total: { currency_code:'EUR', value:sub.toFixed(2) },
              shipping:   { currency_code:'EUR', value:ship.toFixed(2) }
            }
          }
        }]
      }),
      onShippingChange: (d, a) => {
        const ppLand = d.shipping_address?.country_code || 'DE';
        const rawSub  = cart.reduce((acc,i) => acc+i.price*i.qty, 0);
        const ppDisc  = calcDiscount(rawSub);
        const ppDiscSub = rawSub - ppDisc;
        const newShip = getShipping(ppLand, ppDiscSub);
        const sel = document.getElementById('land-sel');
        if(sel) { const o=sel.querySelector(`option[value="${ppLand}"]`); sel.value = o?ppLand:'WORLD'; updateCart(); }
        return a.order.patch([{
          op:'replace',
          path:"/purchase_units/@reference_id=='default'/amount",
          value:{ currency_code:'EUR', value:(ppDiscSub+newShip).toFixed(2), breakdown:{item_total:{currency_code:'EUR',value:ppDiscSub.toFixed(2)},shipping:{currency_code:'EUR',value:newShip.toFixed(2)}} }
        }]);
      },
      onApprove: (d, a) => a.order.capture().then(details => {
        const email   = details.payer?.email_address || '';
        const fn      = details.payer?.name?.given_name || 'Kunde';
        const ln      = details.payer?.name?.surname || '';
        const orderNr = 'LXRD-PP-' + Date.now().toString().slice(-6);

        showOrderSuccess(email, orderNr);
      }),
      onError: () => toast('PayPal Fehler. Erneut versuchen.','err')
    }).render('#pp-btn-cont');
  } catch(e) { console.warn('PayPal render:', e); }
